config firewall policy

    edit 1
        set name "CLIENT_TO_ADC_HTTP"

        set srcintf "port3"
        set dstintf "port2"

        set srcaddr "all"
        set dstaddr "all"

        set action accept
        set schedule "always"
        set service "HTTP"

        set nat disable
    next

        set dstintf "port2"

        set dstaddr "all"

        set action accept
        set schedule "always"
        set service "HTTPS"

        set nat disable
    next

end
